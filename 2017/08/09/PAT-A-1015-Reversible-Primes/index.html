#include <bits/stdc++.h>
using namespace std;

bool isPrime(int s)
{
    if(s < 2)
        return false;
    if(s==2 || s==3)
        return true;
    for(int i=2;i<=sqrt(s);i++)
        if(s%i==0)
            return false;
    return true;
}

int rev_num(int num1,int radix)
{
    int num2 = 0;
    while(num1)
    {
        num2 *= radix;
        num2 += num1 % radix;
        num1 /= radix;
    }
    return num2;
}

int main(void)
{
    int n, d;
    while(scanf("%d", &n))
    {
        if(n<0)
            break;
        scanf("%d", &d);

        int rev= rev_num(n,d);

        if(isPrime(n) && isPrime(rev))
            cout<<"Yes"<<endl;
        else
            cout<<"No"<<endl;
    }
    return 0;
}
