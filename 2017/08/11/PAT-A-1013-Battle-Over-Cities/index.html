题目分析：

图搜索问题，并查集或者DFS，第一遍选了较为熟悉的dfs。DFS解法即为深度遍历全部点，遍历情况记录入visited节点，极端情况中：可从第一点遍历全部其他点，则不用遍历其他点，不用修复任何边；无法从任一点到达任一其他点，则需要全部修复边。并查集方法 WIP。

源代码：

#include <bits/stdc++.h>

using namespace std;

#define MAXN 1000

int ma[MAXN][MAXN];
int vst[MAXN];
int n, m, k;

void dfs(int c)
{
    vst[c]=1;
    for(int i=1;i<=n;i++)
    {
        if(!vst[i] && ma[c][i])
        {
            dfs(i);
        }
    }
}

int main(void)
{
    cin >> n >> m >> k;
    memset(ma, 0, sizeof(ma));
    for (int i = 0;i < m; i++)
    {
        int t1, t2;
        cin >> t1 >> t2;
        ma[t1][t2] = 1;
        ma[t2][t1] = 1;
    }
    int cnt = 0;
    while (k--)
    {
        int temp;
        cnt=0;
        memset(vst, 0, sizeof(vst));
        cin >> temp;
        vst[temp] = 1;
        for (int i = 1; i <= n; i++)
        {
            if (!vst[i])
            {
                dfs(i);
                cnt++;
            }
        }
        cout << cnt-1 << endl;
    }
    return 0;
}
